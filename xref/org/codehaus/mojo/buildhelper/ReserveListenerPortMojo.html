52  /**
53   * Reserve a list of random and not in use network ports and place them in a configurable project properties.
54   *
55   * @author <a href="dantran@gmail.com">Dan T. Tran</a>
56   * @version $Id: ReserveListnerPortMojo.java 6754 2008-04-13 15:14:04Z dantran $
57   * @since 1.2
58   */
59  @Mojo( name = "reserve-network-port", defaultPhase = LifecyclePhase.PROCESS_TEST_CLASSES, threadSafe = true )
60  public class ReserveListenerPortMojo
61      extends AbstractMojo
62  {
63      private static final String BUILD_HELPER_RESERVED_PORTS = "BUILD_HELPER_MIN_PORT";
64  
65      private static final int FIRST_NON_ROOT_PORT_NUMBER = 1024;
66  
67      private static final Integer MAX_PORT_NUMBER = 65535;
68  
69      private static final Object lock = new Object();
70  
71      /**
72       * A List to property names to be placed in Maven project. At least one of {@code #urls} or {@code #portNames} has
73       * to be specified.
74       *
75       * @since 1.2
76       */
77      @Parameter
78      private String[] portNames = new String[0];
79  
80      /**
81       * A List of urls to resource where list of name could be found. Can be in form of classpath:com/myproject/names.txt
82       * . At least one of {@code #urls} or {@code #portNames} has to be specified.
83       * 
84       * @since 1.11
85       */
86      @Parameter
87      private String[] urls = new String[0];
88  
89      /**
90       * Output file to write the generated properties to. if not given, they are written to Maven project
91       *
92       * @since 1.2
93       */
94      @Parameter
95      private File outputFile;
96  
97      /**
98       * Specify this if you want the port be chosen with a number higher than that one.
99       * <p>
100      * If {@link #maxPortNumber} is specified, defaults to {@value #FIRST_NON_ROOT_PORT_NUMBER}.
101      * </p>
102      *
103      * @since 1.8
104      */
105     @Parameter
106     private Integer minPortNumber;
107 
108     /**
109      * Specify this if you want the port be chosen with a number lower than that one.
110      *
111      * @since 1.8
112      */
113     @Parameter
114     private Integer maxPortNumber;
115 
116     /**
117      * Specify true or false if you want the port selection randomized.
118      *
119      * @since 1.10
120      */
121     @Parameter
122     private boolean randomPort;
123 
124     /**
125      * @since 1.2
126      */
127     @Parameter( readonly = true, defaultValue = "${project}" )
128     private MavenProject project;

130     @Override
131     public void execute()
132         throws MojoExecutionException
133     {
134         Properties properties = project.getProperties();
135 
136         if ( outputFile != null )
137         {
138             properties = new Properties();
139         }
140 
141         loadUrls();
142 
143         // Reserve the entire block of ports to guarantee we don't get the same port twice
144         final List<ServerSocket> sockets = new ArrayList<ServerSocket>();
145         try
146         {
147             for ( String portName : portNames )
148             {
149                 try
150                 {
151                     final ServerSocket socket = getServerSocket();
152                     sockets.add( socket );
153 
154                     final String unusedPort = Integer.toString( socket.getLocalPort() );
155                     properties.put( portName, unusedPort );
156                     getReservedPorts().add( socket.getLocalPort() );
157                     this.getLog().info( "Reserved port " + unusedPort + " for " + portName );
158                 }
159                 catch ( IOException e )
160                 {
161                     throw new MojoExecutionException( "Error getting an available port from system", e );
162                 }
163             }
164 
165             // Write the file -- still hold onto the ports
166             if ( outputFile != null )
167             {
168                 try
169                 {
170                     createOutputDirectoryIfNotExist( outputFile );
171                 }
172                 catch ( IOException e )
173                 {
174                     throw new MojoExecutionException( e.getMessage() );
175                 }
176 
177                 OutputStream os = null;
178                 try
179                 {
180                     os = new FileOutputStream( outputFile );
181                     properties.store( os, null );
182                 }
183                 catch ( Exception e )
184                 {
185                     throw new MojoExecutionException( e.getMessage() );
186                 }
187                 finally
188                 {
189                     IOUtil.close( os );
190                 }
191             }
192         }
193         finally
194         {
195             // Now free all the ports
196             for ( ServerSocket socket : sockets )
197             {
198                 final int localPort = socket.getLocalPort();
199                 try
200                 {
201                     socket.close();
202                 }
203                 catch ( IOException e )
204                 {
205                     this.getLog().error( "Cannot free reserved port " + localPort );
206                 }
207             }
208         }
209     }
210 
211     private void createOutputDirectoryIfNotExist( File outputFile )
212         throws IOException
213     {
214         File parentDirectory = new File( outputFile.getCanonicalFile().getParent() );
215 
216         if ( !parentDirectory.exists() )
217         {
218             getLog().debug( "Trying to create directories: " + parentDirectory.getAbsolutePath() );
219             if ( !parentDirectory.mkdirs() )
220             {
221                 getLog().error( "Failed to create folders " + parentDirectory.getAbsolutePath() );
222             }
223         }
224 
225     }

227     private ServerSocket getServerSocket()
228         throws IOException, MojoExecutionException
229     {
230         if ( minPortNumber == null && maxPortNumber != null )
231         {
232             getLog().debug( "minPortNumber unspecified: using default value " + FIRST_NON_ROOT_PORT_NUMBER );
233             minPortNumber = FIRST_NON_ROOT_PORT_NUMBER;
234         }
235         if ( minPortNumber != null && maxPortNumber == null )
236         {
237             getLog().debug( "maxPortNumber unspecified: using default value " + MAX_PORT_NUMBER );
238             maxPortNumber = MAX_PORT_NUMBER;
239         }
240         if ( minPortNumber == null && maxPortNumber == null )
241         {
242             return new ServerSocket( 0 );
243         }
244         if ( randomPort )
245         {
246             synchronized ( lock )
247             {
248                 List<Integer> availablePorts = randomPortList();
249                 for ( Iterator<Integer> iterator = availablePorts.iterator(); iterator.hasNext(); )
250                 {
251                     int port = iterator.next();
252                     ServerSocket serverSocket = reservePort( port );
253                     iterator.remove();
254                     if ( serverSocket != null )
255                     {
256                         return serverSocket;
257                     }
258                 }
259                 throw new MojoExecutionException( "Unable to find an available port between " + minPortNumber + " and "
260                     + maxPortNumber );
261             }
262         }
263         else
264         {
265             // Might be synchronizing a bit too largely, but at least that defensive approach should prevent
266             // threading issues (essentially possible while put/getting the plugin ctx to get the reserved ports).
267             synchronized ( lock )
268             {
269                 int min = getNextPortNumber();
270 
271                 for ( int port = min;; ++port )
272                 {
273 
274                     ServerSocket serverSocket = reservePort( port );
275                     if ( serverSocket != null )
276                     {
277                         return serverSocket;
278                     }
279                 }
280             }
281         }
282     }

284     private List<Integer> randomPortList()
285     {
286 
287         int difference = maxPortNumber - minPortNumber + 1;
288         List<Integer> portList = new ArrayList<Integer>( difference );
289         List<Integer> reservedPorts = getReservedPorts();
290         for ( int i = 0; i < difference; i++ )
291         {
292             int port = minPortNumber + i;
293             if ( !reservedPorts.contains( port ) )
294             {
295                 portList.add( minPortNumber + i );
296             }
297         }
298         Collections.shuffle( portList );
299         return portList;
300     }

302     public ServerSocket reservePort( int port )
303         throws MojoExecutionException
304     {
305 
306         if ( port > maxPortNumber )
307         {
308             throw new MojoExecutionException( "Unable to find an available port between " + minPortNumber + " and "
309                 + maxPortNumber );
310         }
311         try
312         {
313             ServerSocket serverSocket = new ServerSocket( port );
314             getLog().info( "Port assigned " + port );
315             return serverSocket;
316         }
317         catch ( IOException ioe )
318         {
319             getLog().info( "Tried binding to port " + port + " without success. Trying next port.", ioe );
320         }
321         return null;
322     }
323 
324     private int getNextPortNumber()
325     {
326         assert minPortNumber != null;
327 
328         List<Integer> reservedPorts = getReservedPorts();
329         int nextPort = -1;
330         if ( reservedPorts.isEmpty() )
331         {
332             nextPort = minPortNumber;
333         }
334         else
335         {
336             nextPort = findAvailablePortNumber( minPortNumber, reservedPorts );
337         }
338         reservedPorts.add( nextPort );
339         getLog().debug( "Next port: " + nextPort );
340         return nextPort;
341     }
342 
343     @SuppressWarnings( "unchecked" )
344     private List<Integer> getReservedPorts()
345     {
346 
347         List<Integer> reservedPorts = (List<Integer>) getPluginContext().get( BUILD_HELPER_RESERVED_PORTS );
348         if ( reservedPorts == null )
349         {
350             reservedPorts = new ArrayList<Integer>();
351             getPluginContext().put( BUILD_HELPER_RESERVED_PORTS, reservedPorts );
352         }
353         return reservedPorts;
354     }

356     /**
357      * Returns the first number available, starting at portNumberStartingPoint that's not already in the reservedPorts
358      * list.
359      *
360      * @param portNumberStartingPoint first port number to start from.
361      * @param reservedPorts the ports already reserved.
362      * @return first number available not in the given list, starting at the given parameter.
363      */
364     private int findAvailablePortNumber( Integer portNumberStartingPoint, List<Integer> reservedPorts )
365     {
366         assert portNumberStartingPoint != null;
367         int candidate = portNumberStartingPoint;
368         while ( reservedPorts.contains( candidate ) )
369         {
370             candidate++;
371         }
372         return candidate;
373     }

375     private void loadUrls()
376         throws MojoExecutionException
377     {
378         for ( String url : urls )
379         {
380             load( new UrlResource( url ) );
381         }
382     }

384     private void load( UrlResource resource )
385         throws MojoExecutionException
386     {
387         if ( resource.canBeOpened() )
388         {
389             loadPortNamesFromResource( resource );
390         }
391         else
392         {
393             throw new MojoExecutionException( "Port names could not be loaded from \"" + resource + "\"" );
394         }
395     }
396 
397     private void loadPortNamesFromResource( UrlResource resource )
398         throws MojoExecutionException
399     {
400         try
401         {
402             if ( getLog().isDebugEnabled() )
403             {
404                 getLog().debug( "Loading port names from " + resource );
405             }
406             final InputStream stream = resource.getInputStream();
407 
408             try
409             {
410                 BufferedReader reader = new BufferedReader( new InputStreamReader( stream ) );
411                 List<String> names = new ArrayList<String>();
412                 String line;
413                 while ( ( line = reader.readLine() ) != null )
414                 {
415                     line = line.trim();
416                     if ( !line.isEmpty() && !line.replace( " ", "" ).startsWith( "#" ) )
417                     {
418                         names.add( line );
419                     }
420                 }
421                 if ( getLog().isDebugEnabled() )
422                 {
423                     getLog().debug( "Loaded port names " + names );
424                 }
425                 String[] tPortNames = names.toArray( new String[portNames.length + names.size()] );
426                 if ( portNames.length > 0 )
427                 {
428                     System.arraycopy( portNames, 0, tPortNames, names.size(), portNames.length );
429                 }
430                 portNames = tPortNames;
431             }
432             finally
433             {
434                 stream.close();
435             }
436         }
437         catch ( IOException e )
438         {
439             throw new MojoExecutionException( "Error reading port names from \"" + resource + "\"", e );
440         }
441     }

443     private class UrlResource
444     {
445         private static final String CLASSPATH_PREFIX = "classpath:";
446 
447         private static final String SLASH_PREFIX = "/";
448 
449         private final URL url;
450 
451         private boolean isMissingClasspathResouce = false;
452 
453         private String classpathUrl;
454 
455         private InputStream stream;
456 
457         public UrlResource( String url )
458             throws MojoExecutionException
459         {
460             if ( url.startsWith( CLASSPATH_PREFIX ) )
461             {
462                 String resource = url.substring( CLASSPATH_PREFIX.length(), url.length() );
463                 if ( resource.startsWith( SLASH_PREFIX ) )
464                 {
465                     resource = resource.substring( 1, resource.length() );
466                 }
467                 this.url = getClass().getClassLoader().getResource( resource );
468                 if ( this.url == null )
469                 {
470                     if ( getLog().isDebugEnabled() )
471                     {
472                         getLog().debug( "Can not load classpath resouce \"" + url + "\"" );
473                     }
474                     isMissingClasspathResouce = true;
475                     classpathUrl = url;
476                 }
477             }
478             else
479             {
480                 try
481                 {
482                     this.url = new URL( url );
483                 }
484                 catch ( MalformedURLException e )
485                 {
486                     throw new MojoExecutionException( "Badly formed URL " + url + " - " + e.getMessage() );
487                 }
488             }
489         }
490 
491         public InputStream getInputStream()
492             throws IOException
493         {
494             if ( stream == null )
495             {
496                 stream = openStream();
497             }
498             return stream;
499         }
500 
501         public boolean canBeOpened()
502         {
503             if ( isMissingClasspathResouce )
504             {
505                 return false;
506             }
507             try
508             {
509                 openStream().close();
510             }
511             catch ( IOException e )
512             {
513                 return false;
514             }
515             return true;
516         }
517 
518         private InputStream openStream()
519             throws IOException
520         {
521             return new BufferedInputStream( url.openStream() );
522         }
523 
524         @Override
525         public String toString()
526         {
527             if ( !isMissingClasspathResouce )
528             {
529                 return "URL " + url.toString();
530             }
531             return classpathUrl;
532         }
